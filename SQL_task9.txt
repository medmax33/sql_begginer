SELECT t1.ContactName, t1.Region, t2.ContactName, t2.Region
FROM Customers t1, Customers t2 
WHERE (t1.CustomerID <> t2.CustomerID) AND 
(t1.Region = ANY (SELECT Region FROM Customers)) AND
(t2.Region = ANY (SELECT Region FROM Customers));

9.4.2. 
SELECT * FROM Orders t1
WHERE t1.CustomerID =  ANY
(SELECT CustomerID FROM Customers t2 
WHERE t2.Region = ANY (SELECT Region FROM Customers))

9.4.3. 
SELECT * FROM Orders t1
WHERE t1.OrderID =  ANY
(SELECT t2.OrderID FROM [Order Details] t2 
WHERE t1.Freight > t2.UnitPrice)
